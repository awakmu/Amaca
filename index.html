Example

Template file (example.tmpl):

<!DOCTYPE html>

<head>
    <title>{{ return title }}</title>
</head>

<body>
    <h1>{{ return body }}</h1>
    {{
        output = ''
        for i = 10, 1, -1 do
            output = output .. '\n\t\t<p>Number ' .. i .. '</p>\n'
        end

        return output
    }}
</body>
</html>

C code (example.c):

#include <stdio.h>
#include "amaca.h"

int main() {
    char *str = Amaca_template_file(
        "example.tmpl", 2,
        "title", "This is a title",
        "body", "This is the body"
    );

    printf("%s", str);

    return 0;
}
